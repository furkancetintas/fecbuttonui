## Install

```bash
npm install --save fecbuttonui
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'fecbuttonui'
import 'fecbuttonui/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
